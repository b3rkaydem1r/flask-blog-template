create table users (id int(11) not null auto_increment primary key, name varchar(64) not null, username varchar(64) not null, email varchar(64) not null, authority int(11) not null default 1, password text not null);
create table posts (id int(11) not null auto_increment primary key, title varchar(150) not null, author varchar(64) not null, content text not null, created_date DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP);
create table comments (id int(11) not null auto_increment primary key, username varchar(64) not null, comment varchar(600) not null, post_id int(11) not null, status int(11) not null default 0);
